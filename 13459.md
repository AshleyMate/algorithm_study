# [13459](https://www.acmicpc.net/problem/13459)

```
from sys import stdin
from collections import deque
input = stdin.readline

n, m = map(int, input().split())
matrix = [list(input().strip()) for _ in range(n)]
check = [[[[False]*m for _ in range(n)] for _ in range(m)] for _ in range(n)]
dx, dy = [-1, 0, 1, 0], [0, 1, 0, -1]
q = deque()


def init():
    rx, ry, bx, by = [0]*4
    for i in range(n):
        for j in range(m):
            if matrix[i][j] == 'R':
                rx, ry = i, j
            elif matrix[i][j] == 'B':
                bx, by = i, j
    q.append((rx, ry, bx, by, 0)) #빨간 구슬과 파란 구슬의 위치가 q에 담긴다
    check[rx][ry][bx][by] = True


def move(x, y, dx, dy, count):
    while matrix[x + dx][y + dy] != '#' and matrix[x][y] != 'O':
        x += dx
        y += dy
        count += 1
    return x, y, count


def bfs():
    while q:
        rx, ry, bx, by, count = q.popleft()
        if count >= 10:
            break
        for i in range(4):
            nrx, nry, rcnt = move(rx, ry, dx[i], dy[i], 0)
            nbx, nby, bcnt = move(bx, by, dx[i], dy[i], 0)
            if matrix[nbx][nby] == 'O':
                continue
            if matrix[nrx][nry] == 'O':
                print(1)
                return
            if nrx == nbx and nry == nby:
                if rcnt > bcnt:
                    nrx, nry = nrx-dx[i], nry-dy[i]
                else:
                    nbx, nby = nbx-dx[i], nby-dy[i]
            if not check[nrx][nry][nbx][nby]:
                check[nrx][nry][nbx][nby] = True
                q.append((nrx, nry, nbx, nby, count+1))
    print(0)


init()
bfs()

```
